CREATE TABLE User (
id	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
name	TEXT UNIQUE,
email	TEXT UNIQUE,
created_at	REAL,
last_online	REAL,
pw 	TEXT);

CREATE TABLE GameCharacter (
id	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
user_id	INTEGER,
name	TEXT,
xp	INTEGER,
lv	INTEGER,
posX	INTEGER,
posY	INTEGER,
inventorySize	INTEGER,
inventoryMaxSize	INTEGER,
FOREIGN KEY(user_id) REFERENCES User(id));

CREATE TABLE Item (
id	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
name TEXT NOT NULL UNIQUE,
type TEXT NOT NULL);

CREATE TABLE InventoryItems (
gcID	INTEGER NOT NULL,
itemID	INTEGER NOT NULL,
FOREIGN KEY(itemID) REFERENCES Item(id),
FOREIGN KEY(gcID) REFERENCES GameCharacter(id));
